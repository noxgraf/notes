cat domains.txt | waybackurls > urls

Auto​

  • King of Bug Bounty Tips
# basics
nikto -h http://<domain>

nuclei --target http://<domain>

# with headers and follow redirect
nuclei --target http://<domain> -header "Cookie: connect.sid=;" -follow-redirects -max-redirects 10 -verbose

# full template
nuclei -l alive.txt -t /nuclei-templates/ -severity critical,high,medium -c 50 -rl 150 -o nuclei_results.txt

# CVEs
nuclei -l alive.txt -t cves/ -severity critical,high -c 30 -o cve_results.txt

# subfinder to nuclei
subfinder -d example.com -silent | httpx -silent | nuclei -severity critical,high -c 50 -o results.txt

# mass API Fuzzing
cat api_endpoints.txt | httpx -silent -mc 200,201,401,403 | nuclei -dast -t dast/vulnerabilities/ -H "Content-Type: application/json" -rl 20 -c 5 -o api_nuclei_dast.txt

# exposed panels only
nuclei -l alive.txt -t exposed-panels/ -c 50 | anew panels.txt

# secrets from JS
cat js.txt | httpx -silent -sr -srd js_files/ && nuclei -t exposures/ -target js.txt

# API keys from JS
cat js.txt | nuclei -t exposures/tokens/ -silent | anew api_keys.txt

DNS​

Types:

  • A Record: Hostname to IPv4.
  • AAAA Record: Hostname to IPv6.
  • CNAME Record: Canonical Name - domain to another domain.
  • MX Record: Mail Exchange - mail server responsible for handling emails for the domain.
  • TXT: Text Records: arbitrary text, commonly used for SPF, DKIM, DMARC, and domain verification.
  • SOA: Start of Authority: the primary name server, admin email, and zone serial number.

Basics​

# fast
host <domain>

# legacy
nslookup -type=<dns_type> <domain> <dns> 

# dig
dig @<dns> <domain> <dns_type>

DNS Bruteforce​

gobuster dns -d "<domain>" -w /usr/share/wordlists/SecLists/Discovery/DNS/<list>

# -m -> match | -f -> filter | -ac -> autofilter
ffuf -u "FUZZ.<domain>" -w /usr/share/wordlists/SecLists/Discovery/DNS/<list> -t 60 -ac

Amass​

amass enum -passive -d <target>

amass enum -d <target>

Vhost Bruteforce​

gobuster vhost -u "http://<ip>" --domain "<domain>" --append-domain -t 120 -w /usr/share/wordlists/SecLists/Discovery/DNS/<list> -r -s 200 --exclude-length <default_length>

ffuf -u 'https://<ip>' -H "Host: FUZZ.<domain>" -w /usr/share/wordlists/SecLists/Discovery/DNS/<list> -t 60 -ac

Directory Bruteforce​

gobuster dir -u "http://<domain>" -w /usr/share/wordlists/SecLists/Discovery/Web-Content/<list> -x php,asp,txt,xml,html,bak,zip,7z,rar,old -r -s 200, 204, 301, 302, 307, 403

ffuf -u "https://<domain>/FUZZ" -w /usr/share/wordlists/SecLists/Discovery/Web-Content/<list> -t 60 -ac

ffuf -u "http://<domain>/FUZZ"  -w /usr/share/wordlists/SecLists/Discovery/Web-Content/<list> -e php,asp,txt,xml,html,bak,zip,7z,rar,old -mc 200,301,302

# reduced waf noise
ffuf -u "https://<target>/FUZZ" -w /usr/share/wordlists/SecLists/Discovery/Web-Content/<wordlist> -H "User-Agent: Mozilla/5.0" -H "X-Forwarded-For: 127.0.0.1" -t 1 -p 1 -mc all
