Pointers​

int a, *p; 

int *p 
p = null; // null pointer

int i = 10;
printf("%p", &i); // mem address of i

int *p; // int-pointer
p = &i; // p is address-of i
*p = 20; // derreference, indirection (*), i is now 20


void increment(int *p){
	*p = *p + 1; // add one to where p points to
}

int main(void) {
	int i = 10; 
	int *j = &i; // int-pointer to address-of i
	
	increment(j); // j is an int-pointer to i
	// increment(&i);  do the same
	printf("i is %d\n", i);
	
	return 0;
}

Math​

a = a + 3; // add, a += 3;
a = a - 3; // sub, a -= 3;
a = a * 3; // mult, a*= 3;
a = a / 3; // div, a /= 3;
a = a % 3; // modulo, a %= 3;
a++; // add one, post increment, a + b++; compute a+b, then +1
a--; // sub one, post increment
++a; // add one, pre increment, a + ++b; increment 1 to be, then a + b
--a; // sub one, pre increment
// exponent -> pow(), math.h

Loop​

// for (initialize things; loop if this is true; do this after each loop)
// for (i = 0, j = 999; i < 10; i++, j--)
// for(;;) infinite
for(i = 0; i < 10; i++){
	printf("i is %d\n", i);
}

// when i == 10, stop
while ( i < 10) {
	printf("loop");
	i++;
}

// when i == 10, last pass
do {
	printf("true")
} while (i < 10);

while (1) {
	printf("infinite loop");
	break; // pass; or continue;
}
