Sockets​

#!/usr/bin/env python3
import sys
import socket

if len(sys.argv) == 2:
	target = socket.gethostbyname(sys.argv[1]) # argv 0 is command.py
else:
	print("Missing args.")
	print("Usage: python3 scanner.py <ip>")

try:
	for port in range(50,85) # example ports
		s = socket.socket(socket.AF_INET, socket.SOCK_STREAM) #ipv4, tcp
		socket.setdefaulttimeout(1) # 1 sec, else move on
		result = s.connect_ex((target,port)) # connect with exception, 0 connected
			if result == 0:
				print(f"Port {port} is open")
			s.close()
						
except KeyboardInterrupt:
	print("end")
	sys.exit()
		
except socket.gaierror:
	print("hostname not resolved.")
	sys.exit()
		
except socket.error:
    print("couldn't connecet")
	sys.exit()

ETH​

#!/usr/bin/env python3
import socket as s

MAC_SRC=b"\x11\x11\x11\x11\x11\x11"
MAC_DST=b"\x22\x22\x22\x22\x22\x22"
ETH_TYPE=b"\xff\xff"

eth = s.socket(s.AF_PACKET, s.SOCK_RAW, int.from_bytes(ETH_TYPE, byteorder="big"))
eth.bind(("eth0", 0))

frame = MAC_SRC + MAC_DST + ETH_TYPE

eth.send(frame)

class Books:

	def __init__(self, name, author, genre, price, stock):
		self.name = name
		self.author = author
		self.genre = genre
		self.price = float(price)
		self.stock= stock
	def in_stock(self):
		if self.stock >= 1:
			return True
		else
			return False
		
From Books import Books

book1 = Books("A study in scarlet" , "Arthur Connan Doyle", "Romance", 20, 5)

print(book1.genre) # Romance
print(book1.in_stock()) # True
